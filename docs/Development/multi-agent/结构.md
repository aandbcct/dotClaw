agentRun边界

- 如果中间需要**释放上下文、等待外部输入、或切换执行主体**，就应该切分为新的 AgentRun。
- 如果只是同一个 LLM 调用内部的 Function Calling 循环（模型自己决定调工具并继续生成），通常视为**同一个 AgentRun 的内部步骤**。

```python
# 1. Session（减负）
@dataclass
class Session: # 连续会话容器
    id: str
    title: str
    turns: list[Conversation]   # 不再是 messages: list[SessionMessage]
    
@dataclass
class Conversation: # 单次请求包含的所有对话
    user_message: str
    assistant_response: str
    agent_runs: list[str]           # request_id 列表

# 2. AgentRun → 显式建模（原本的 _history）
@dataclass
class AgentRun: # 单次用户请求的所有消息
    request_id: str
    agent_id: str
    parent_run_id: str | None       # 子 AgentRun 指向父
    messages: list[Message]         # ReAct 完整消息链
    status: "running|completed|failed"
    spawned_runs: list[str]         # 子 AgentRun 的 request_id 列表

# 3. Agent 不再直接持有 Session/Conversation
class Agent:
    identity: AgentIdentity
    runtime: AgentRuntime
    current_run: AgentRun | None    # 当前正在执行的 AgentRun
    conversation: Conversation | None  # 当前对话（父 Agent 才有）
```

```python
# ── AgentRuntime: 纯能力，纯执行 ──
# "我给你一个 LLM 调用、给你一个工具执行器、给你一套 Skill"
# 不关心你是谁、有什么权限、用哪个 model
runtime = AgentRuntime(
    llm=llm_proxy,              # "我能调模型"
    tool_executor=tool_executor, # "我能执行工具（全部注册的）"
    memory_mgr=memory_mgr,      # "我能读写记忆"
    skill_registry=skill_registry, # "我有这些 Skill 可用"
    mcp_provider=mcp_provider,  # "我有这些 MCP 工具"
    assembler=assembler,        # "我能组装上下文"
)

# ── AgentIdentity: 纯声明，纯约束 ──
# "用哪些工具、叫什么名字、system prompt 是什么"
# 不持有任何可执行的东西
identity = AgentIdentity(
    agent_id="security-reviewer",
    agent_name="安全审查员",          # ← 影响 system prompt 中的自称
    allowed_tools=["read_file", "exec"], # ← 限制工具白名单
    system_prompt_template="你是{agent_name}，专注于...", # ← 行为指令
    model="deepseek-v4-pro",          # ← 用哪个模型
    max_loop_steps=20,                # ← 最多跑几轮
)

class Agent:
    identity: AgentIdentity      # 声明
    runtime: AgentRuntime        # 能力

    def _resolve_tool_definitions(self) -> list[ToolDefinition]:
        # Identity 约束 Runtime
        all_defs = self.runtime.tool_executor.get_definitions()  # Runtime 提供全部
        allowed = self.identity.allowed_tools                     # Identity 做过滤
        if not allowed:
            return all_defs
        return [d for d in all_defs if d.name in set(allowed)]

    async def _invoke_llm(self, messages, tool_definitions, journal):
        # Identity 指定模型，Runtime 执行调用
        model = self.identity.model or self._config.llm.default_model  # Identity
        async for chunk in self.runtime.llm.chat(...):                  # Runtime
            ...

    def _resolve_system_prompt(self) -> str:
        # Identity 提供模板，Runtime 不参与
        return self.identity.system_prompt_template.format(
            agent_name=self.identity.agent_name,
            workspace=self.identity.workspace,
        )
```
